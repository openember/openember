project(web_dashboard VERSION 1.0.0 LANGUAGES CXX)

set( WEB_MODULE_SRCS 
    main.cpp
)

set(_web_root "${OPENEMBER_WEB_DASHBOARD_ROOT_DIR}")
if(NOT _web_root OR _web_root STREQUAL "")
    set(_web_root "${CMAKE_CURRENT_SOURCE_DIR}/web_root")
elseif(NOT IS_ABSOLUTE "${_web_root}")
    set(_web_root "${CMAKE_SOURCE_DIR}/${_web_root}")
endif()

OPENEMBER_ADD_EXECUTABLE(TARGET ${PROJECT_NAME} 
    SRCS     ${WEB_MODULE_SRCS} 
    LIBS     ${OPENEMBER_LIBRARIES} ${MONGOOSE_LIBRARIES}
    INCLUDES ${OPENEMBER_INCLUDE_DIRS} ${MONGOOSE_INCLUDE_DIRS}
    DESTINATION bin 
)

target_compile_definitions(${PROJECT_NAME} PRIVATE
    OPENEMBER_WEB_DASHBOARD_ROOT_DIR="${_web_root}"
)

if(NOT DEFINED OPENEMBER_WEB_DASHBOARD_PORT OR OPENEMBER_WEB_DASHBOARD_PORT STREQUAL "")
    set(OPENEMBER_WEB_DASHBOARD_PORT 8000)
endif()
target_compile_definitions(${PROJECT_NAME} PRIVATE
    OPENEMBER_WEB_DASHBOARD_PORT=${OPENEMBER_WEB_DASHBOARD_PORT}
)

if(NOT DEFINED OPENEMBER_LOGGER_PORT OR OPENEMBER_LOGGER_PORT STREQUAL "")
    set(OPENEMBER_LOGGER_PORT 18081)
endif()
target_compile_definitions(${PROJECT_NAME} PRIVATE
    OPENEMBER_LOGGER_PORT=${OPENEMBER_LOGGER_PORT}
)

if(NOT DEFINED OPENEMBER_SPDLOG_ENABLE_TOPIC)
    set(OPENEMBER_SPDLOG_ENABLE_TOPIC OFF)
endif()
if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_NAME OR OPENEMBER_SPDLOG_TOPIC_NAME STREQUAL "")
    set(OPENEMBER_SPDLOG_TOPIC_NAME "/openember/log")
endif()
if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_SUB_URL OR OPENEMBER_SPDLOG_TOPIC_SUB_URL STREQUAL "")
    if(OPENEMBER_PUBSUB_BACKEND STREQUAL "lcm")
        set(OPENEMBER_SPDLOG_TOPIC_SUB_URL "udpm://239.255.76.67:7667?ttl=1")
    elseif(OPENEMBER_PUBSUB_BACKEND STREQUAL "udp")
        set(OPENEMBER_SPDLOG_TOPIC_SUB_URL "udp://127.0.0.1:7560")
    else()
        set(OPENEMBER_SPDLOG_TOPIC_SUB_URL "tcp://127.0.0.1:7561")
    endif()
endif()
target_compile_definitions(${PROJECT_NAME} PRIVATE
    OPENEMBER_SPDLOG_ENABLE_TOPIC=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_TOPIC}>,1,0>
    OPENEMBER_SPDLOG_TOPIC_NAME="${OPENEMBER_SPDLOG_TOPIC_NAME}"
    OPENEMBER_SPDLOG_TOPIC_SUB_URL="${OPENEMBER_SPDLOG_TOPIC_SUB_URL}"
)

file(COPY web_root DESTINATION "${CMAKE_CURRENT_BINARY_DIR}/../..")
