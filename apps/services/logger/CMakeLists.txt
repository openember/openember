project(logger VERSION 1.0.0 LANGUAGES CXX)

set(LOGGER_MODULE_SRCS
    main.cpp
)

OPENEMBER_ADD_EXECUTABLE(TARGET ${PROJECT_NAME}
    SRCS     ${LOGGER_MODULE_SRCS}
    LIBS     ${OPENEMBER_LIBRARIES} ${MONGOOSE_LIBRARIES}
    INCLUDES ${OPENEMBER_INCLUDE_DIRS} ${MONGOOSE_INCLUDE_DIRS}
    DESTINATION bin
)

set_property(TARGET ${PROJECT_NAME} PROPERTY CXX_STANDARD 17)
set_property(TARGET ${PROJECT_NAME} PROPERTY CXX_STANDARD_REQUIRED ON)

if(NOT DEFINED OPENEMBER_LOGGER_PORT OR OPENEMBER_LOGGER_PORT STREQUAL "")
    set(OPENEMBER_LOGGER_PORT 18081)
endif()
if(NOT DEFINED OPENEMBER_LOGGER_LOG_DIR OR OPENEMBER_LOGGER_LOG_DIR STREQUAL "")
    set(OPENEMBER_LOGGER_LOG_DIR "${OPENEMBER_SPDLOG_FILE_DIR}")
endif()

target_compile_definitions(${PROJECT_NAME} PRIVATE
    OPENEMBER_LOGGER_PORT=${OPENEMBER_LOGGER_PORT}
    OPENEMBER_LOGGER_LOG_DIR="${OPENEMBER_LOGGER_LOG_DIR}"
)
