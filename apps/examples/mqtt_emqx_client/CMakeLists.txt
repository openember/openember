# 连接公共 EMQX 示例 Broker（或自建）；参数由 Kconfig/genconfig 或 CMake 覆盖。
if(NOT DEFINED OPENEMBER_MQTT_EMQX_BROKER_URI OR "${OPENEMBER_MQTT_EMQX_BROKER_URI}" STREQUAL "")
    set(OPENEMBER_MQTT_EMQX_BROKER_URI "tcp://broker.emqx.io:1883")
endif()
if(NOT DEFINED OPENEMBER_MQTT_EMQX_CLIENT_ID OR "${OPENEMBER_MQTT_EMQX_CLIENT_ID}" STREQUAL "")
    set(OPENEMBER_MQTT_EMQX_CLIENT_ID "openember-mqtt-example")
endif()
if(NOT DEFINED OPENEMBER_MQTT_EMQX_USERNAME)
    set(OPENEMBER_MQTT_EMQX_USERNAME "")
endif()
if(NOT DEFINED OPENEMBER_MQTT_EMQX_PASSWORD)
    set(OPENEMBER_MQTT_EMQX_PASSWORD "")
endif()
if(NOT DEFINED OPENEMBER_MQTT_EMQX_TOPIC OR "${OPENEMBER_MQTT_EMQX_TOPIC}" STREQUAL "")
    set(OPENEMBER_MQTT_EMQX_TOPIC "emqx/c-test")
endif()

OPENEMBER_ADD_EXECUTABLE(TARGET mqtt_emqx_client
    SRCS     main.cpp
    LIBS     ${OPENEMBER_LIBRARIES}
    INCLUDES ${OPENEMBER_INCLUDE_DIRS}
    DESTINATION bin
)

target_compile_definitions(mqtt_emqx_client PRIVATE
    "OPENEMBER_MQTT_EMQX_BROKER_URI=\"${OPENEMBER_MQTT_EMQX_BROKER_URI}\""
    "OPENEMBER_MQTT_EMQX_CLIENT_ID=\"${OPENEMBER_MQTT_EMQX_CLIENT_ID}\""
    "OPENEMBER_MQTT_EMQX_USERNAME=\"${OPENEMBER_MQTT_EMQX_USERNAME}\""
    "OPENEMBER_MQTT_EMQX_PASSWORD=\"${OPENEMBER_MQTT_EMQX_PASSWORD}\""
    "OPENEMBER_MQTT_EMQX_TOPIC=\"${OPENEMBER_MQTT_EMQX_TOPIC}\""
)
