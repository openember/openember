project(Log VERSION 1.0.0)

set(LIB_LOG_SRCS
    log_wrapper.c
    oe_log_c.c
)
if(OPENEMBER_LOG_BACKEND STREQUAL "SPDLOG")
    list(APPEND LIB_LOG_SRCS log_spdlog.cpp)
endif()

OPENEMBER_ADD_STATIC_LIBRARY(TARGET ${PROJECT_NAME}
    SRCS     ${LIB_LOG_SRCS}
    LIBS     ${OPENEMBER_EXTERNAL_LIBRARIES} msgbus
    INCLUDES ${OPENEMBER_EXTERNAL_INCLUDE_DIRS} ${OPENEMBER_INTERNAL_INCLUDE_DIRS} ${CMAKE_CURRENT_SOURCE_DIR}
    DESTINATION lib
)

if(OPENEMBER_LOG_BACKEND STREQUAL "SPDLOG")
    set_property(TARGET Log PROPERTY CXX_STANDARD 17)
    set_property(TARGET Log PROPERTY CXX_STANDARD_REQUIRED ON)

    if(NOT DEFINED OPENEMBER_SPDLOG_LEVEL OR "${OPENEMBER_SPDLOG_LEVEL}" STREQUAL "")
        set(OPENEMBER_SPDLOG_LEVEL "info")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_FLUSH_LEVEL OR "${OPENEMBER_SPDLOG_FLUSH_LEVEL}" STREQUAL "")
        set(OPENEMBER_SPDLOG_FLUSH_LEVEL "info")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_PATTERN OR "${OPENEMBER_SPDLOG_PATTERN}" STREQUAL "")
        set(OPENEMBER_SPDLOG_PATTERN "[%Y-%m-%d %H:%M:%S.%e] [%^%l%$] %v")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_ENABLE_STDOUT)
        set(OPENEMBER_SPDLOG_ENABLE_STDOUT ON)
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_ENABLE_FILE)
        set(OPENEMBER_SPDLOG_ENABLE_FILE ON)
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_FILE_DIR OR "${OPENEMBER_SPDLOG_FILE_DIR}" STREQUAL "")
        set(OPENEMBER_SPDLOG_FILE_DIR "/var/log/openember")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_ROTATE_MAX_MB OR "${OPENEMBER_SPDLOG_ROTATE_MAX_MB}" STREQUAL "")
        set(OPENEMBER_SPDLOG_ROTATE_MAX_MB 10)
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_ROTATE_MAX_FILES OR "${OPENEMBER_SPDLOG_ROTATE_MAX_FILES}" STREQUAL "")
        set(OPENEMBER_SPDLOG_ROTATE_MAX_FILES 5)
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_ENABLE_SYSLOG)
        set(OPENEMBER_SPDLOG_ENABLE_SYSLOG OFF)
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_ENABLE_TOPIC)
        set(OPENEMBER_SPDLOG_ENABLE_TOPIC ON)
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_NAME OR "${OPENEMBER_SPDLOG_TOPIC_NAME}" STREQUAL "")
        set(OPENEMBER_SPDLOG_TOPIC_NAME "/openember/log")
    endif()
    # Topic bind/connect URLs follow the selected msgbus backend (LCM vs NNG).
    if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_PUB_URL OR "${OPENEMBER_SPDLOG_TOPIC_PUB_URL}" STREQUAL "")
        if(OPENEMBER_MSGBUS_USE_LCM)
            set(OPENEMBER_SPDLOG_TOPIC_PUB_URL "udpm://239.255.76.67:7667?ttl=1")
        else()
            set(OPENEMBER_SPDLOG_TOPIC_PUB_URL "tcp://*:7561")
        endif()
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_SUB_URL OR "${OPENEMBER_SPDLOG_TOPIC_SUB_URL}" STREQUAL "")
        if(OPENEMBER_MSGBUS_USE_LCM)
            set(OPENEMBER_SPDLOG_TOPIC_SUB_URL "udpm://239.255.76.67:7667?ttl=1")
        else()
            set(OPENEMBER_SPDLOG_TOPIC_SUB_URL "tcp://127.0.0.1:7561")
        endif()
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_LEVEL OR "${OPENEMBER_SPDLOG_TOPIC_LEVEL}" STREQUAL "")
        set(OPENEMBER_SPDLOG_TOPIC_LEVEL "info")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_TOPIC_RATE_LIMIT OR "${OPENEMBER_SPDLOG_TOPIC_RATE_LIMIT}" STREQUAL "")
        set(OPENEMBER_SPDLOG_TOPIC_RATE_LIMIT 0)
    endif()

    target_compile_definitions(Log PRIVATE
        OPENEMBER_SPDLOG_LEVEL="${OPENEMBER_SPDLOG_LEVEL}"
        OPENEMBER_SPDLOG_FLUSH_LEVEL="${OPENEMBER_SPDLOG_FLUSH_LEVEL}"
        OPENEMBER_SPDLOG_PATTERN="${OPENEMBER_SPDLOG_PATTERN}"
        OPENEMBER_SPDLOG_ENABLE_STDOUT=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_STDOUT}>,1,0>
        OPENEMBER_SPDLOG_ENABLE_FILE=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_FILE}>,1,0>
        OPENEMBER_SPDLOG_FILE_DIR="${OPENEMBER_SPDLOG_FILE_DIR}"
        OPENEMBER_SPDLOG_ROTATE_MAX_MB=${OPENEMBER_SPDLOG_ROTATE_MAX_MB}
        OPENEMBER_SPDLOG_ROTATE_MAX_FILES=${OPENEMBER_SPDLOG_ROTATE_MAX_FILES}
        OPENEMBER_SPDLOG_ENABLE_SYSLOG=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_SYSLOG}>,1,0>
        OPENEMBER_SPDLOG_ENABLE_TOPIC=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_TOPIC}>,1,0>
        OPENEMBER_SPDLOG_TOPIC_NAME="${OPENEMBER_SPDLOG_TOPIC_NAME}"
        OPENEMBER_SPDLOG_TOPIC_PUB_URL="${OPENEMBER_SPDLOG_TOPIC_PUB_URL}"
        OPENEMBER_SPDLOG_TOPIC_SUB_URL="${OPENEMBER_SPDLOG_TOPIC_SUB_URL}"
        OPENEMBER_SPDLOG_TOPIC_LEVEL="${OPENEMBER_SPDLOG_TOPIC_LEVEL}"
        OPENEMBER_SPDLOG_TOPIC_RATE_LIMIT=${OPENEMBER_SPDLOG_TOPIC_RATE_LIMIT}
    )
endif()
