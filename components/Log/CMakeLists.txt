project(Log VERSION 1.0.0)

set(LIB_LOG_SRCS
    log_wrapper.c
    oe_log_c.c
)
if(OPENEMBER_LOG_BACKEND STREQUAL "SPDLOG")
    list(APPEND LIB_LOG_SRCS log_spdlog.cpp)
endif()

OPENEMBER_ADD_STATIC_LIBRARY(TARGET ${PROJECT_NAME}
    SRCS     ${LIB_LOG_SRCS}
    LIBS     ${OPENEMBER_EXTERNAL_LIBRARIES}
    INCLUDES ${OPENEMBER_EXTERNAL_INCLUDE_DIRS} ${OPENEMBER_INTERNAL_INCLUDE_DIRS} ${CMAKE_CURRENT_SOURCE_DIR}
    DESTINATION lib
)

if(OPENEMBER_LOG_BACKEND STREQUAL "SPDLOG")
    set_property(TARGET Log PROPERTY CXX_STANDARD 17)
    set_property(TARGET Log PROPERTY CXX_STANDARD_REQUIRED ON)

    if(NOT DEFINED OPENEMBER_SPDLOG_LEVEL OR "${OPENEMBER_SPDLOG_LEVEL}" STREQUAL "")
        set(OPENEMBER_SPDLOG_LEVEL "info")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_FLUSH_LEVEL OR "${OPENEMBER_SPDLOG_FLUSH_LEVEL}" STREQUAL "")
        set(OPENEMBER_SPDLOG_FLUSH_LEVEL "info")
    endif()
    if(NOT DEFINED OPENEMBER_SPDLOG_PATTERN OR "${OPENEMBER_SPDLOG_PATTERN}" STREQUAL "")
        set(OPENEMBER_SPDLOG_PATTERN "[%Y-%m-%d %H:%M:%S.%e] [%^%l%$] %v")
    endif()

    target_compile_definitions(Log PRIVATE
        OPENEMBER_SPDLOG_LEVEL="${OPENEMBER_SPDLOG_LEVEL}"
        OPENEMBER_SPDLOG_FLUSH_LEVEL="${OPENEMBER_SPDLOG_FLUSH_LEVEL}"
        OPENEMBER_SPDLOG_PATTERN="${OPENEMBER_SPDLOG_PATTERN}"
        OPENEMBER_SPDLOG_ENABLE_STDOUT=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_STDOUT}>,1,0>
        OPENEMBER_SPDLOG_ENABLE_FILE=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_FILE}>,1,0>
        OPENEMBER_SPDLOG_FILE_DIR="${OPENEMBER_SPDLOG_FILE_DIR}"
        OPENEMBER_SPDLOG_ROTATE_MAX_MB=${OPENEMBER_SPDLOG_ROTATE_MAX_MB}
        OPENEMBER_SPDLOG_ROTATE_MAX_FILES=${OPENEMBER_SPDLOG_ROTATE_MAX_FILES}
        OPENEMBER_SPDLOG_ENABLE_SYSLOG=$<IF:$<BOOL:${OPENEMBER_SPDLOG_ENABLE_SYSLOG}>,1,0>
    )
endif()
